def simulate_rolling_die(trials=15600):        #TASK: Modify the number of trials and the situation
    rolls_greater_than_4 = sum([1 for _ in range(trials) if random.randint(1, 6) > 4])
    probability = rolls_greater_than_4 / trials

import random

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def simulate_picking_prime(trials=11000):  #TASK: Modify the number of trials and the situation
    prime_picks = sum([1 for _ in range(trials) if is_prime(random.randint(1, 10))])
    probability = prime_picks / trials
    return probability

print(simulate_picking_prime())

# Include these libraries
import numpy as np
from itertools import permutations
import matplotlib.pyplot as plt
import networkx as nx

# Plot a graph represented by an adjacency_matrix
def plot_graph(adjacency_matrix):
    """
    Visualizes a graph given its adjacency matrix.

    Parameters:
    adjacency_matrix (np.ndarray): The adjacency matrix of the graph to visualize.
    """
    # Converts the adjacency matrix into a NetworkX graph object for visualization.
    G = nx.from_numpy_array(adjacency_matrix)
    
    # Sets the figure size for the plot.
    plt.figure(figsize=(4, 4))
    
    # Draws the graph with specified aesthetics.
    nx.draw(G, with_labels=True, node_color='skyblue', node_size=700, edge_color='k', linewidths=2, font_size=15, pos=nx.spring_layout(G))
    
    # Sets the title of the plot.
    plt.title('Graph Visualization')
    
    # Displays the plot.
    plt.show()

# Check if two graphs are isomorphic
def are_isomorphic(G1, G2):
    """
    Checks if two graphs represented by their adjacency matrices are isomorphic.

    Parameters:
    G1, G2 (np.ndarray): Adjacency matrices of the two graphs to compare.

    Returns:
    bool: True if the graphs are isomorphic, False otherwise.
    """
    # Checks if the matrices are square and of the same size, a prerequisite for isomorphism.
    if G1.shape != G2.shape or G1.shape[0] != G1.shape[1]:
        return False

    # The number of vertices in the graphs.
    n = G1.shape[0]
    
    # Iterates over all possible permutations of vertex labels.
    for perm in permutations(range(n)):
        # Applies the current permutation to rows and columns of G1 to simulate vertex relabeling.
        permuted_G1 = G1[np.ix_(perm, perm)]
        
        # Checks if the permuted version of G1 is identical to G2, indicating isomorphism.
        if np.array_equal(permuted_G1, G2):
            return True
    
    # Returns False if no permutation resulted in identical adjacency matrices, indicating the graphs are not isomorphic.
    return False
